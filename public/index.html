Quick Start Snippets

Copy-paste into your project and start using the API instantly.

cURL
# Random quote
curl https://pinoy-dev-quotes-api.vercel.app/api/v1/random

# Search
curl "https://pinoy-dev-quotes-api.vercel.app/api/v1/search?q=bug"

# Plain text format
curl "https://pinoy-dev-quotes-api.vercel.app/api/v1/random?format=text"
JavaScript
// Fetch a random quote
const res = await fetch(
  'https://pinoy-dev-quotes-api.vercel.app/api/v1/random'
);
const { data } = await res.json();
console.log(`"${data.quote}" — ${data.author}`);
Python
import requests

r = requests.get(
    "https://pinoy-dev-quotes-api.vercel.app/api/v1/random"
)
quote = r.json()["data"]
print(f'"{quote["quote"]}" — {quote["author"]}')
PHP
$json = file_get_contents(
    "https://pinoy-dev-quotes-api.vercel.app/api/v1/random"
);
$quote = json_decode($json, true)["data"];
echo "\"{$quote['quote']}\" — {$quote['author']}";
GraphQL
// POST to /api/v1/graphql
{
  quotes(dialect: "bisaya", limit: 3) {
    id
    quote
    language
    english_translation
    author
    tags
  }
}
